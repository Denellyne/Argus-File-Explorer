#include <filesystem>


namespace fs = std::filesystem;
class Drive {
public:
    Drive();
    const std::filesystem::space_info driveBytes;
    uintmax_t driveSize;
    std::string filePath;
    uintmax_t fileSize;

};
class File : public Drive {
public:
    File();
    void newPath(std::string newPath);
    void getSize();
private:
    const double toMegabyte;
};
class Directory : public Drive {
public:
    Directory();
    void newPath(std::string newPath);
};
